## 알고리즘) 1~100 합과 곱

```js
let s = 0;
let m = 1;
let n = 100;
for (var i = 1; i < n + 1; i++) {
  s += i;
  // s = s+i;
}
// O(n)
console.log((n * (n + 1)) / 2);
//O(1)
console.log(s);

function f(n) {
  if (n <= 1) {
    return 1;
  } else {
    return n + f(n - 1);
  }
}
console.log(f(100), '재귀함수');

for (var i = 1; i < 6; i++) {
  m *= i;
  // s = s*i;
}
console.log(m);
```
